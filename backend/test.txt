package main

import "fmt"

func main() {
	numbers := []int{10, 20, 30, 40, 50}

	// Scenario 1: Iterate only using the value, ignore the index
	fmt.Println("Iterating only over values:")
	for _, value := range numbers {
		fmt.Printf("Value: %d\n", value)
	}

	// Scenario 2: Iterate using both index and value
	fmt.Println("\nIterating over values with index:")
	for idx, value := range numbers {
		fmt.Printf("Index: %d, Value: %d\n", idx, value)
	}
}
